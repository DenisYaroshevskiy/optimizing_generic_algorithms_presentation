std::merge


template <typename I1,typename I2, typename O, typename P>
O merge(I1 f1, I1 l1, I2 f2, I2 l2, O o, P p) {
  for (; f1 != l1; ++o) {
    if (f2 == l2) return std::copy(f1, l1, o);

    if (p(*f2, *f1)) {
      *o = *f2; ++f2;
    } else {
      *o = *f1; ++f1;
    }
  }

  return std::copy(f2, l2, o);
}
